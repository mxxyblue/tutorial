CREATE TABLE "CALBOARD" 
   (	"CSEQ" NUMBER NOT NULL ENABLE, 
	"ID" VARCHAR2(40) NOT NULL ENABLE, 
	"CALTITLE" VARCHAR2(100) NOT NULL ENABLE, 
	"CALCONTENT" VARCHAR2(4000) NOT NULL ENABLE, 
	"MDATE" VARCHAR2(12) NOT NULL ENABLE, 
	"CALREGDATE" DATE NOT NULL ENABLE, 
	 CONSTRAINT "CALBOARD_PK_cseq" PRIMARY KEY ("CSEQ")
	 CONSTRAINT "FK_USERINFO_id" FOREIGN KEY ("ID")
	 REFERENCES USERINFO" ("ID") ENABLE
   );
   
CREATE SEQUENCE calboard_seq INCREMENT BY 1 START WITH 1;
