CREATE TABLE MEMBER 
(
  ID VARCHAR2(40) NOT NULL 
, NAME VARCHAR2(100) NOT NULL 
, PASSWORD VARCHAR2(100) NOT NULL 
, ADDRESS VARCHAR2(1000) NOT NULL 
, PHONE VARCHAR2(50) NOT NULL 
, EMAIL VARCHAR2(100) NOT NULL 
, ENABLED VARCHAR2(2) NOT NULL 
, ROLE VARCHAR2(10) NOT NULL 
, REGDATE DATE NOT NULL 
, CONSTRAINT MEMBER_PK PRIMARY KEY 
  (
    ID
  )
  ENABLE 
);

ALTER TABLE MEMBER
ADD CONSTRAINT EMAIL_UNIQUE UNIQUE 
(
  EMAIL 
)
ENABLE;

CREATE TABLE TEAM 
(
  TEAMNUM NUMBER NOT NULL 
, TUTOR VARCHAR2(20) NOT NULL 
, TUTEE1 VARCHAR2(20) NOT NULL 
, TUTEE2 VARCHAR2(20) 
, TUTEE3 VARCHAR2(20) 
, TUTEE4 VARCHAR2(20) 
, CONSTRAINT TEAM_PK PRIMARY KEY 
  (
    TEAMNUM 
  )
  ENABLE 
);

CREATE SEQUENCE TEAM_SEQ INCREMENT BY 1 START WITH 1 MINVALUE 1;
